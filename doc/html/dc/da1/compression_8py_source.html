96def prediction(ld_vector, weight_vector, x, y, z, data):
97 # Eq(36) Calculate inner product of local difference and weight vectors
98 d = np.inner(weight_vector, ld_vector)
99
100 # Eq(37) Calculate high resolution predicted sample values, broken into temp vars
101 temp1 = d + (2**W_RES) * (local_sum(x, y, z, data) - 4 * S_MID)
102 temp2 = (temp1 + (2**(R - 1))) % (2**R) - 2**(R - 1)
103 temp3 = temp2 + (2**(W_RES + 2) * S_MID) + 2**(W_RES + 1)
104
105 min = 2**(W_RES + 2) * S_MIN
106 max = 2**(W_RES + 2) * S_MAX + 2**(W_RES + 1)
107 hr_s = np.clip(temp3, min, max)
108
109 # Eq(38) Calculate double-resolution predicted sample value
110 Nx = data.shape[0]
111 t = y * Nx + x
112 s_tilde = 0
113
114 if t > 0:
115 s_tilde = np.floor(hr_s / (2**(W_RES + 1)))
116 elif t == 0 and P > 0 and z > 0:
117 s_tilde = 2 * data[x, y, z - 1]
118 elif t == 0 and (P == 0 or z == 0):
119 s_tilde = 2 * S_MID
120
121 # Eq(39)
122 s_hat = np.floor(s_tilde / 2)
123
124 return s_hat, s_tilde,d, hr_s

212def GPO2(k, j):

214 #Write unary code
