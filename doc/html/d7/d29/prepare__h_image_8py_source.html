31def wavelength_to_rgb(wavelength, gamma=0.8):
32
33 '''This converts a given wavelength of light to an
34 approximate RGB color value. The wavelength must be given
35 in nanometers in the range from 380 nm through 750 nm
36 (789 THz through 400 THz).
37 Based on code by Dan Bruton
38 http://www.physics.sfasu.edu/astro/color/spectra.html
39 '''
40
41 wavelength = float(wavelength)
42 if wavelength >= 380 and wavelength <= 440:
43 attenuation = 0.3 + 0.7 * (wavelength - 380) / (440 - 380)
44 R = ((-(wavelength - 440) / (440 - 380)) * attenuation) ** gamma
45 G = 0.0
46 B = (1.0 * attenuation) ** gamma
47 elif wavelength >= 440 and wavelength <= 490:
48 R = 0.0
49 G = ((wavelength - 440) / (490 - 440)) ** gamma
50 B = 1.0
51 elif wavelength >= 490 and wavelength <= 510:
52 R = 0.0
53 G = 1.0
54 B = (-(wavelength - 510) / (510 - 490)) ** gamma
55 elif wavelength >= 510 and wavelength <= 580:
56 R = ((wavelength - 510) / (580 - 510)) ** gamma
57 G = 1.0
58 B = 0.0
59 elif wavelength >= 580 and wavelength <= 645:
60 R = 1.0
61 G = (-(wavelength - 645) / (645 - 580)) ** gamma
62 B = 0.0
63 elif wavelength >= 645 and wavelength <= 750:
64 attenuation = 0.3 + 0.7 * (750 - wavelength) / (750 - 645)
65 R = (1.0 * attenuation) ** gamma
66 G = 0.0
67 B = 0.0
68 else:
69 R = 0.0
70 G = 0.0
71 B = 0.0
72 R *= 255
73 G *= 255
74 B *= 255
75 return (int(R), int(G), int(B))
