9def generate_perlin_noise_3d(
10 shape, res, tileable=(False, False, False),
11 interpolant=interpolant
12):
13 """Generate a 3D numpy array of perlin noise.
14
15 Args:
16 shape: The shape of the generated array (tuple of three ints).
17 This must be a multiple of res.
18 res: The number of periods of noise to generate along each
19 axis (tuple of three ints). Note shape must be a multiple
20 of res.
21 tileable: If the noise should be tileable along each axis
22 (tuple of three bools). Defaults to (False, False, False).
23 interpolant: The interpolation function, defaults to
24 t*t*t*(t*(t*6 - 15) + 10).
25
26 Returns:
27 A numpy array of shape shape with the generated noise.
28
29 Raises:
30 ValueError: If shape is not a multiple of res.
31 """
32 delta = (res[0] / shape[0], res[1] / shape[1], res[2] / shape[2])
33 d = (shape[0] // res[0], shape[1] // res[1], shape[2] // res[2])
34 grid = np.mgrid[0:res[0]:delta[0],0:res[1]:delta[1],0:res[2]:delta[2]]
35 grid = np.mgrid[0:res[0]:delta[0],0:res[1]:delta[1],0:res[2]:delta[2]]
36 grid = grid.transpose(1, 2, 3, 0) % 1
37 # Gradients
38 theta = 2*np.pi*np.random.rand(res[0] + 1, res[1] + 1, res[2] + 1)
39 phi = 2*np.pi*np.random.rand(res[0] + 1, res[1] + 1, res[2] + 1)
40 gradients = np.stack(
41 (np.sin(phi)*np.cos(theta), np.sin(phi)*np.sin(theta), np.cos(phi)),
42 axis=3
43 )
44 if tileable[0]:
45 gradients[-1,:,:] = gradients[0,:,:]
46 if tileable[1]:
47 gradients[:,-1,:] = gradients[:,0,:]
48 if tileable[2]:
49 gradients[:,:,-1] = gradients[:,:,0]
50 gradients = gradients.repeat(d[0], 0).repeat(d[1], 1).repeat(d[2], 2)
51 g000 = gradients[ :-d[0], :-d[1], :-d[2]]
52 g100 = gradients[d[0]: , :-d[1], :-d[2]]
53 g010 = gradients[ :-d[0],d[1]: , :-d[2]]
54 g110 = gradients[d[0]: ,d[1]: , :-d[2]]
55 g001 = gradients[ :-d[0], :-d[1],d[2]: ]
56 g101 = gradients[d[0]: , :-d[1],d[2]: ]
57 g011 = gradients[ :-d[0],d[1]: ,d[2]: ]
58 g111 = gradients[d[0]: ,d[1]: ,d[2]: ]
59 # Ramps
60 n000 = np.sum(np.stack((grid[:,:,:,0] , grid[:,:,:,1] , grid[:,:,:,2] ), axis=3) * g000, 3)
61 n100 = np.sum(np.stack((grid[:,:,:,0]-1, grid[:,:,:,1] , grid[:,:,:,2] ), axis=3) * g100, 3)
62 n010 = np.sum(np.stack((grid[:,:,:,0] , grid[:,:,:,1]-1, grid[:,:,:,2] ), axis=3) * g010, 3)
63 n110 = np.sum(np.stack((grid[:,:,:,0]-1, grid[:,:,:,1]-1, grid[:,:,:,2] ), axis=3) * g110, 3)
64 n001 = np.sum(np.stack((grid[:,:,:,0] , grid[:,:,:,1] , grid[:,:,:,2]-1), axis=3) * g001, 3)
65 n101 = np.sum(np.stack((grid[:,:,:,0]-1, grid[:,:,:,1] , grid[:,:,:,2]-1), axis=3) * g101, 3)
66 n011 = np.sum(np.stack((grid[:,:,:,0] , grid[:,:,:,1]-1, grid[:,:,:,2]-1), axis=3) * g011, 3)
67 n111 = np.sum(np.stack((grid[:,:,:,0]-1, grid[:,:,:,1]-1, grid[:,:,:,2]-1), axis=3) * g111, 3)
68 # Interpolation
69 t = interpolant(grid)
70 n00 = n000*(1-t[:,:,:,0]) + t[:,:,:,0]*n100
71 n10 = n010*(1-t[:,:,:,0]) + t[:,:,:,0]*n110
72 n01 = n001*(1-t[:,:,:,0]) + t[:,:,:,0]*n101
73 n11 = n011*(1-t[:,:,:,0]) + t[:,:,:,0]*n111
74 n0 = (1-t[:,:,:,1])*n00 + t[:,:,:,1]*n10
75 n1 = (1-t[:,:,:,1])*n01 + t[:,:,:,1]*n11
76 return ((1-t[:,:,:,2])*n0 + t[:,:,:,2]*n1)

79def generate_fractal_noise_3d(
80 shape, res, octaves=1, persistence=0.5, lacunarity=2,
81 tileable=(False, False, False), interpolant=interpolant
82):
83 """Generate a 3D numpy array of fractal noise.
84
85 Args:
86 shape: The shape of the generated array (tuple of three ints).
87 This must be a multiple of lacunarity**(octaves-1)*res.
88 res: The number of periods of noise to generate along each
89 axis (tuple of three ints). Note shape must be a multiple of
90 (lacunarity**(octaves-1)*res).
91 octaves: The number of octaves in the noise. Defaults to 1.
92 persistence: The scaling factor between two octaves.
93 lacunarity: The frequency factor between two octaves.
94 tileable: If the noise should be tileable along each axis
95 (tuple of three bools). Defaults to (False, False, False).
96 interpolant: The, interpolation function, defaults to
97 t*t*t*(t*(t*6 - 15) + 10).
98
99 Returns:
100 A numpy array of fractal noise and of shape shape generated by
101 combining several octaves of perlin noise.
102
103 Raises:
104 ValueError: If shape is not a multiple of
105 (lacunarity**(octaves-1)*res).
106 """
107 noise = np.zeros(shape)
108 frequency = 1
109 amplitude = 1
110 for _ in range(octaves):
111 noise += amplitude * generate_perlin_noise_3d(
112 shape,
113 (frequency*res[0], frequency*res[1], frequency*res[2]),
114 tileable,
115 interpolant
116 )
117 frequency *= lacunarity
118 amplitude *= persistence
119 return noise
